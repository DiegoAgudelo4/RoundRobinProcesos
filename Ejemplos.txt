def llenar_procesos():

    for i in range(num_procesos):

        num_ie = int(input(f'Ingrese la cantidad de E/S del proceso [{nombre}]: '))
        lista_ie = [obtener_ie() for _ in range(num_ie)]

            'tiempo_llegada': tiempo_llegada,
